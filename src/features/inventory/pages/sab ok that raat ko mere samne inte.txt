Step 3: Watchdog Script Create + Start
Bash

adb shell "cat > /data/local/tmp/silentknox.sh << 'SCRIPT'
#!/system/bin/sh
while true; do
  am force-stop com.samsung.android.kgclient 2>/dev/null
  am force-stop com.sec.enterprise.knox.cloudmdm.smdms 2>/dev/null
  am force-stop com.samsung.klmsagent 2>/dev/null
  killall com.samsung.android.kgclient 2>/dev/null
  killall com.sec.enterprise.knox.cloudmdm.smdms 2>/dev/null
  killall com.samsung.klmsagent 2>/dev/null
  sleep 0.2
done
SCRIPT"
adb shell chmod +x /data/local/tmp/silentknox.sh
adb shell "nohup sh /data/local/tmp/silentknox.sh > /dev/null 2>&1 &"
